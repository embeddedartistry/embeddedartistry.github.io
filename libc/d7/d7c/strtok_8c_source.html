43 char* __strtok_r(char* s, const char* delim, char** last)
44 {
45  char *spanp, *tok;
46  int c, sc;
47 
48  if(s == NULL && (s = *last) == NULL)
49  return (NULL);
50 
51 /*
52  * Skip (span) leading delimiters (s += strspn(s, delim), sort of).
53  */
54 cont:
55  c = *s++;
56  for(spanp = (char*)delim; (sc = *spanp++) != 0;)
57  {
58  if(c == sc)
59  goto cont;
60  }
61 
62  if(c == 0)
63  { /* no non-delimiter characters */
64  *last = NULL;
65  return (NULL);
66  }
67  tok = s - 1;
68 
69  /*
70  * Scan token (scan for delimiters: s += strcspn(s, delim), sort of).
71  * Note that delim must have one NUL; we stop if we see that, too.
72  */
73  for(;;)
74  {
75  c = *s++;
76  spanp = (char*)delim;
77  do
78  {
79  if((sc = *spanp++) == c)
80  {
81  if(c == 0)
82  s = NULL;
83  else
84  s[-1] = '\0';
85  *last = s;
86  return (tok);
87  }
88  } while(sc != 0);
89  }
90  /* NOTREACHED */
91 }

93 char* strtok(char* s, const char* delim)
94 {
95  static char* last;
96 
97  return (__strtok_r(s, delim, &last));
98 }
