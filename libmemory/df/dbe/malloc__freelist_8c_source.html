79 {

119 void free(void* ptr)
120 {
121  alloc_node_t *blk, *free_blk;
122 
123  // Don't free a NULL pointer..
124  if(ptr)
125  {
126  // we take the pointer and use container_of to get the corresponding alloc block
127  blk = container_of(ptr, alloc_node_t, block);
128 
129  // Let's put it back in the proper spot
130  list_for_each_entry(free_blk, &free_list, node)
131  {
132  if(free_blk > blk)
133  {
134  list_add_(&blk->node, free_blk->node.prev, &free_blk->node);
135  goto blockadded;
136  }
137  }
138  list_add_tail(&blk->node, &free_list);
139 
140  blockadded:
141  // Let's see if we can combine any memory
142  defrag_free_list();
143  }
144 }
